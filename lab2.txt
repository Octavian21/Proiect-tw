2)

afisati numele concatenat cu prenumele unui student la intamplare
set serveroutput on;
DECLARE
      p_std_random INT;
      p_m INT;
      p_rez VARCHAR2(100);
BEGIN
      select count(*) into p_m FROM studenti;
      
      p_std_random := dbms_random.value(1,p_m);
      
      select nume || ' ' || prenume into p_rez
      from
          (select nume,prenume,rownum as rand
              from studenti)
      where rand = p_std_random;
      
      DBMS_OUTPUT.PUT_LINE(p_rez);  
END;

GASITI persoana sau una din persoane cu numele cel mai lung din tabela studenti. afisati prima litera cu majuscula, iar celelalte mici si lungimea numelui

set serveroutput on;
DECLARE
     p_nume studenti.nume%type;
     p_prenume studenti.prenume%type; 
BEGIN
      
      select nume, prenume
      into p_nume,p_prenume
      from
        (select nume,prenume from studenti 
            order by length(trim(nume)) desc )
      where rownum=1;
      
      DBMS_OUTPUT.PUT_LINE(initcap(p_nume)||' '||length(trim(p_nume)));  
END;
